105 {

109 
110  if ( (RUUVI_INTERFACE_COMMUNICATION_RADIO_BEFORE == evt) &&
111  ( (ruuvi_interface_rtc_millis() - last_sample) >
112  APPLICATION_ADC_SAMPLE_INTERVAL_MS))
113  {
114  err_code |= task_adc_vdd_prepare();
115  triggered = (RUUVI_DRIVER_SUCCESS == err_code);
116  }
117 
118  if ( (RUUVI_INTERFACE_COMMUNICATION_RADIO_AFTER == evt) &&
119  triggered)
120  {
121  err_code |= task_adc_vdd_sample();
122  triggered = false;
123 
124  if (RUUVI_DRIVER_SUCCESS == err_code)
125  {
126  last_sample = ruuvi_interface_rtc_millis();
127  }
128  }
129 
130  RUUVI_DRIVER_ERROR_CHECK (err_code, ~RUUVI_DRIVER_ERROR_FATAL);
131 }

133 /*
134  * @brief Initialize MCU peripherals.
135  *
136  * The watchdog is initialized first.
137  * After watchdog, GPIO, including interrupts, are initialized.
138  * Leds are initialized by activating led pins as high-drive outputs and inactive.
139  * SPI is initialized by reserving a peripheral and configuring I/O modes.
140  * I2C is initialized by reserving a peripheral and configuring I/O modes.
141  *
142  * Timer and RTC are initialized by reserving the peripherals.
143  * Scheduler is SW-only, initialization reserves buffer for task queue.
144  *
145  * Once timer is initialized, sleep is configured to go to deepest possible sleep mode
146  * and wake on timer on instead of busylooping. This causes imprecision to ms delays.
147  *
148  * If DC/DC converter is installed on board, it is initialized.
149  *
150  * Flash is initialized by checking if the filesystem can be allocated. In some cases
151  * such as changing allocated flash filesystem size the filesystem area is erased
152  * and program reboots.
153  *
154  * ADC is not reserved at this stage, but VDD is sampled.
155  *
156  */
157 #ifndef CEEDLING

159 #endif
160 void init_mcu (void)
161 {
162  // Init watchdog
163  ruuvi_interface_watchdog_init (APPLICATION_WATCHDOG_INTERVAL_MS);
164  // Init yield
165  ruuvi_driver_status_t status = ruuvi_interface_yield_init();
166  RUUVI_DRIVER_ERROR_CHECK (status, RUUVI_DRIVER_SUCCESS);
167  // Init GPIO
168  status = task_gpio_init();
169  // Initialize LED gpio pins
170  status |= task_led_init();
171  RUUVI_DRIVER_ERROR_CHECK (status, RUUVI_DRIVER_SUCCESS);
172  // Initialize SPI, I2C
173  status = task_spi_init();
174  status |= task_i2c_init();
175  RUUVI_DRIVER_ERROR_CHECK (status, RUUVI_DRIVER_SUCCESS);
176  // Initialize RTC, timer and scheduler. Enable low-power sleep.
177  // Dummy implementation of RTC requires timer.
178  status = task_timer_init();
179  status |= task_rtc_init();
180  status |= task_scheduler_init();
181  // Low power yield goes to deep sleep and wakes the CPU with a timer,
182  // hence timer is required.
183  status |= ruuvi_interface_yield_low_power_enable (true);
184  RUUVI_DRIVER_ERROR_CHECK (status, RUUVI_DRIVER_SUCCESS);
185  // Initialize power perihperals,
186  status |= task_power_dcdc_init();
187  RUUVI_DRIVER_ERROR_CHECK (status, RUUVI_DRIVER_SUCCESS);
188  RUUVI_DRIVER_ERROR_CHECK (status, RUUVI_DRIVER_SUCCESS);
189  // Initialize flash
190  status = task_flash_init();
191  RUUVI_DRIVER_ERROR_CHECK (status, RUUVI_DRIVER_SUCCESS);
192  // Sample VDD
193  status |= task_adc_vdd_prepare();
194  status |= task_adc_vdd_sample();
195  RUUVI_DRIVER_ERROR_CHECK (status, RUUVI_DRIVER_SUCCESS);
196 }

530 int app_main (void)
531 {
532  ruuvi_driver_status_t err_code = RUUVI_DRIVER_SUCCESS;
533  init_logging(); // Initializes logging to user console
534  run_mcu_tests(); // Runs tests which do not rely on MCU peripherals being initialized
535  init_mcu(); // Initialize MCU peripherals, except for communication with users.
536  // Delay one second to make sure timestamps are > 1 s after initialization
537  ruuvi_interface_delay_ms (BOOT_DELAY_MS);
538  err_code |= task_led_write (RUUVI_BOARD_LED_RED,
539  true); // Turn activity led on
540  run_sensor_tests(); // Run tests which rely on MCU peripherals, e.g. sensor drivers
541  init_sensors(); // Initializes sensors with application-defined default mode.
542  // Initialize communication with outside world - BLE, NFC, Buttons
543  /* IMPORTANT! After this point pausing the program flow asserts,
544  * since softdevice asserts on missed timer. This includes debugger.
545  */
546  // run comms tests - TODO @ojousima
547  init_comms();
548  // Turn activity led off. Turn status_ok led on if no errors occured
549  err_code |= task_led_write (RUUVI_BOARD_LED_RED, false);
550 
551  if (RUUVI_DRIVER_SUCCESS == ruuvi_driver_errors_clear())
552  {
553  err_code |= task_led_write (RUUVI_BOARD_LED_STATUS_OK, true);
554  err_code |= task_led_activity_led_set (RUUVI_BOARD_LED_STATUS_OK);
555  ruuvi_interface_delay_ms (BOOT_DELAY_MS);
556  }
557  else
558  {
559  err_code |= task_led_write (RUUVI_BOARD_LED_STATUS_ERROR, true);
560  err_code |= task_led_activity_led_set (RUUVI_BOARD_LED_STATUS_ERROR);
561  ruuvi_interface_delay_ms (BOOT_DELAY_MS);
562  }
563 
564  // Turn LEDs off
565  err_code |= task_led_write (RUUVI_BOARD_LED_STATUS_OK, false);
566  err_code |= task_led_write (RUUVI_BOARD_LED_STATUS_ERROR, false);
567  // Configure activity indication
568  ruuvi_interface_yield_indication_set (&task_led_activity_indicate);
569  RUUVI_DRIVER_ERROR_CHECK (err_code, RUUVI_DRIVER_SUCCESS);
570 
571  do
572  {
573  // Execute scheduled tasks
574  ruuvi_interface_scheduler_execute();
575  // Sleep - woken up on event
576  ruuvi_interface_yield();
577  } while (LOOP_FOREVER);
578 
579  // Intentionally non-reachable code.
580  return -1;
581 }

54 static void on_error (const ruuvi_driver_status_t err,
55  const bool fatal,
56  const char * file,
57  const int line)
58 {
59  if (!fatal)
60  {
61  return;
62  }
63 
64  error_cause_t error = {.error = err, .line = line };
65  ruuvi_driver_status_t err_code;
66  uint32_t timeout = 0;
67  strncpy (error.filename, file, sizeof (error.filename));
68  // Store reason of fatal error
69  err_code = task_flash_store (APPLICATION_FLASH_ERROR_FILE,
70  APPLICATION_FLASH_ERROR_RECORD,
71  &error, sizeof (error));
72 
73  // Wait for flash store op to complete
74  while (RUUVI_DRIVER_SUCCESS == err_code &&
75  timeout < 1000 &&
76  ruuvi_interface_flash_is_busy())
77  {
78  timeout++;
79  // Use microsecond wait to busyloop instead of millisecond wait to low-power sleep
80  // as low-power sleep may hang on interrupt context.
81  ruuvi_interface_delay_us (1000);
82  }

87 }

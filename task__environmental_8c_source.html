558 static void execute_log (void * event, uint16_t event_size)
559 {
560  static uint32_t tick_count = 0;
561  uint32_t interval_ms = APPLICATION_ENVIRONMENTAL_LOG_INTERVAL_MS;
562  uint32_t tick_interval = APPLICATION_ENVIRONMENTAL_TICK_MS;
563  const uint32_t ticks_per_log = interval_ms / tick_interval;
564 
565  if (0 == tick_count)
566  {
567  task_environmental_log();
568  }
569 
570  tick_count = (ticks_per_log <= tick_count) ? 0 : (tick_count + 1);
571 }

578 ruuvi_driver_status_t task_environmental_init (void)
579 {
580  ruuvi_driver_status_t err_code = RUUVI_DRIVER_SUCCESS;
581  // Attempt to initialize all possible temperature backends.
582  err_code |= initialize_tmp117();
583  err_code |= initialize_shtcx();
584  err_code |= initialize_bme280();
585  err_code |= initialize_ntc();
586  err_code |= initialize_mcu();
587  err_code |= initialize_lis2dh12();
588 
589  // Use first valid backend as the default provider.
590  for (int ii = 0; ii < ENV_SENSOR_COUNT; ii++)
591  {
592  if (m_environmental_sensors[ii].provides.bitfield)
593  {
594  m_active_sensor = & (m_environmental_sensors[ii]);
595  break;
596  }
597  }
598 
599  // XXX - use generic logging
600  if (NULL == m_log_timer)
601  {
602  ruuvi_interface_timer_create (&m_log_timer, RUUVI_INTERFACE_TIMER_MODE_REPEATED,
603  schedule_log);
604  }
605 
606  ruuvi_interface_timer_start (m_log_timer, APPLICATION_ENVIRONMENTAL_TICK_MS);
607  // Log environmental conditions at start
608  ruuvi_interface_scheduler_event_put (NULL, 0, execute_log);
609  return (NULL == m_active_sensor) ? RUUVI_DRIVER_ERROR_NOT_FOUND : RUUVI_DRIVER_SUCCESS;
610 }

643  ruuvi_driver_sensor_data_t data = {0};
644  float values[3];
645  data.data = values;
646  data.fields.datas.temperature_c = 1;
647  data.fields.datas.humidity_rh = 1;
648  data.fields.datas.pressure_pa = 1;
649  ruuvi_driver_status_t err_code = RUUVI_DRIVER_SUCCESS;
650  err_code |= task_environmental_data_get (&data);
651  RUUVI_DRIVER_ERROR_CHECK (err_code, ~RUUVI_DRIVER_ERROR_FATAL);
652  uint64_t sampletime = data.timestamp_ms / 1000;
653  uint32_t logtime = (uint32_t) sampletime;
654  uint64_t systime = task_rtc_millis() / 1000;
655 
656  if (logtime > systime)
657  {
658  LOGW ("Stored sample in the future\r\n");
659  }
660 
661  environmental_log_t log = { .timestamp_s = logtime,
662  .temperature_c = ruuvi_driver_sensor_data_parse (&data, (ruuvi_driver_sensor_data_fields_t)
663  {
664  .datas.temperature_c = 1
665  }),
666  .humidity_rh = ruuvi_driver_sensor_data_parse (&data, (ruuvi_driver_sensor_data_fields_t)
667  {
668  .datas.humidity_rh = 1
669  }),
670  .pressure_pa = ruuvi_driver_sensor_data_parse (&data, (ruuvi_driver_sensor_data_fields_t)
671  {
672  .datas.pressure_pa = 1
673  })
674  };
675  ruuvi_library_status_t status = ruuvi_library_ringbuffer_queue (&ringbuf, &log,
676  sizeof (log));
677 
678  // Drop old sample if buffer is full
679  if (RUUVI_LIBRARY_ERROR_NO_MEM == status)
680  {
681  LOG ("Discarded data... ");
682  environmental_log_t drop;
683  status = ruuvi_library_ringbuffer_dequeue (&ringbuf, &drop);
684  status |= ruuvi_library_ringbuffer_queue (&ringbuf, &log, sizeof (log));
685  }
686 
687  RUUVI_DRIVER_ERROR_CHECK (status, ~RUUVI_DRIVER_ERROR_FATAL);
688 
689  if (RUUVI_DRIVER_SUCCESS == (status | err_code))
690  {
691  LOG ("Stored data\r\n");
692  }
693 
694  return status | err_code;
695 }

697 ruuvi_driver_status_t task_environmental_log_read (const
698  ruuvi_interface_communication_xfer_fp_t reply_fp,
699  const ruuvi_interface_communication_message_t * const query)
700 {
701  LOG ("Preparing to send logs\r\n");
702  ruuvi_interface_communication_message_t msg = { 0 };
703  uint64_t systime = task_rtc_millis() / 1000;
704  uint32_t now = (query->data[3] << 24) +
705  (query->data[4] << 16) +
706  (query->data[5] << 8) +
707  (query->data[6] << 0);
708  uint32_t start = (query->data[7] << 24) +
709  (query->data[8] << 16) +
710  (query->data[9] << 8) +
711  (query->data[10] << 0);
712  uint32_t offset = (now > systime) ? now - systime : 0;
713  environmental_log_t * p_log;
714  msg.data_length = RUUVI_ENDPOINT_STANDARD_MESSAGE_LENGTH;
715  msg.data[0] = query->data[1];
716  msg.data[1] = query->data[0];
717  msg.data[2] = RUUVI_ENDPOINT_STANDARD_LOG_VALUE_WRITE;
718  ruuvi_library_status_t status = RUUVI_LIBRARY_SUCCESS;
719  // As long as we have more elements
720  size_t index = 0;
721 
722  do
723  {
724  status = ruuvi_library_ringbuffer_peek (&ringbuf, &p_log, index++);
725 
726  // Send logged element
727  if (RUUVI_LIBRARY_SUCCESS == status)
728  {
729  // Calculate real time of event
730  uint32_t timestamp = offset + p_log->timestamp_s;
731 
732  // Check if the event is in range to send. Continue if not
733  if (0 == p_log->timestamp_s)
734  {
735  LOG ("WARNING: Empty element\r\n");
736  continue;
737  }
738 
739  if (now - offset < p_log->timestamp_s)
740  {
741  LOG ("WARNING: Element in future\r\n");
742  continue;
743  }
744 
745  if (now - offset - p_log->timestamp_s > (2 * (ringbuf.index_mask *
746  (APPLICATION_ENVIRONMENTAL_LOG_INTERVAL_MS / 1000))))
747  {
748  LOG ("WARNING: Element in distant past\r\n");
749  continue;
750  }
751 
752  if (start > timestamp)
753  {
754  continue;
755  }
756 
757  msg.data[3] = timestamp >> 24;
758  msg.data[4] = timestamp >> 16;
759  msg.data[5] = timestamp >> 8;
760  msg.data[6] = timestamp >> 0;
761  uint8_t destination = query->data[0];
762 
763  // send temp, humi, pressure
764  if (RUUVI_ENDPOINT_STANDARD_DESTINATION_ENVIRONMENTAL == destination ||
765  RUUVI_ENDPOINT_STANDARD_DESTINATION_TEMPERATURE == destination)
766  {
767  int32_t temperature_cc = p_log->temperature_c * 100;
768  msg.data[1] = RUUVI_ENDPOINT_STANDARD_DESTINATION_TEMPERATURE;
769  msg.data[7] = temperature_cc >> 24;
770  msg.data[8] = temperature_cc >> 16;
771  msg.data[9] = temperature_cc >> 8;
772  msg.data[10] = temperature_cc >> 0;
773 
774  // Repeat sending here
775  while (RUUVI_LIBRARY_ERROR_NO_MEM == reply_fp (&msg))
776  {
777  // Sleep
778  ruuvi_interface_yield();
779  }
780  }
781 
782  if (RUUVI_ENDPOINT_STANDARD_DESTINATION_ENVIRONMENTAL == destination ||
783  RUUVI_ENDPOINT_STANDARD_DESTINATION_HUMIDITY == destination)
784  {
785  uint32_t humidity_crh = p_log->humidity_rh * 100;
786  msg.data[1] = RUUVI_ENDPOINT_STANDARD_DESTINATION_HUMIDITY;
787  msg.data[7] = humidity_crh >> 24;
788  msg.data[8] = humidity_crh >> 16;
789  msg.data[9] = humidity_crh >> 8;
790  msg.data[10] = humidity_crh >> 0;
791 
792  // Repeat sending here
793  while (RUUVI_LIBRARY_ERROR_NO_MEM == reply_fp (&msg))
794  {
795  // Sleep
796  ruuvi_interface_yield();
797  }
798  }
799 
800  if (RUUVI_ENDPOINT_STANDARD_DESTINATION_ENVIRONMENTAL == destination ||
801  RUUVI_ENDPOINT_STANDARD_DESTINATION_PRESSURE == destination)
802  {
803  if (isnan (p_log->pressure_pa))
804  {
805  continue;
806  }
807 
808  uint32_t pressure_pa = p_log->pressure_pa;
809  msg.data[1] = RUUVI_ENDPOINT_STANDARD_DESTINATION_PRESSURE;
810  msg.data[7] = pressure_pa >> 24;
811  msg.data[8] = pressure_pa >> 16;
812  msg.data[9] = pressure_pa >> 8;
813  msg.data[10] = pressure_pa >> 0;
814 
815  // Repeat sending here
816  while (RUUVI_LIBRARY_ERROR_NO_MEM == reply_fp (&msg))
817  {
818  // Sleep
819  ruuvi_interface_yield();
820  }
821  }
822  }
823  } while (RUUVI_LIBRARY_SUCCESS == status);
824 
825  LOG ("Logs sent\r\n");
826  return RUUVI_DRIVER_SUCCESS;
827 }
